# GitHub'a Yükleme Komutları

# 1. Git kurulu mu kontrol et
git --version

# 2. Git config (ilk kurulumsa)
git config --global user.name "Adınız Soyadınız"
git config --global user.email "emailiniz@example.com"

# 3. Local repository başlat
git init

# 4. Dosyaları ekle
git add .

# 5. İlk commit
git commit -m "Initial commit - Ulusal Tevkil Ağı Projesi"

# 6. GitHub'da repository oluştur (web'den):
# https://github.com/new
# Repository adı: tevkil_proje
# Public seç
# README ekleme (zaten var)

# 7. Remote ekle (KULLANICIADI'nı değiştir!)
git remote add origin https://github.com/KULLANICIADI/tevkil_proje.git

# 8. Push
git branch -M main
git push -u origin main

# ✅ Artık GitHub'da!
# Link: https://github.com/KULLANICIADI/tevkil_proje
